num_outliers <- function(x) {
  x <- x[!is.na(x)]
  q25 <- quantile(x, 0.25)
  q75 <- quantile(x, 0.75)
  IQR_val <- q75 - q25
  lower_bound <- q25 - 1.5 * IQR_val
  upper_bound <- q75 + 1.5 * IQR_val
  num_outliers <- sum(x < lower_bound | x > upper_bound)
  return(num_outliers)
}

Gen_X_numbers <- function(n) {
  X_vals <- numeric(n)
  for (i in 1:n) {
    u <- runif(1)
    if (u < 0.5) {
      X_vals[i] <- 0
    } else if (u < 0.7) {
      X_vals[i] <- 3
    } else {
      X_vals[i] <- 10
    }
  }
  return(X_vals)
}

  U <- runif(n)
  X <- numeric(n)
  X[U < (1 - alpha - beta)] <- 0
  X[U >= (1 - alpha - beta) & U < (1 - beta)] <- 3

sample_X_0310 <- function(alpha, beta, n) {
  U <- runif(n)
  X <- case_when(
    (U >= 0) & (U < alpha) ~ 3,
    (U >= alpha) & (U < alpha + beta) ~ 10,
    (U >= alpha + beta) & (U <= 1) ~ 0
  )
  return(X)
}

my_cdf_exp <- function(x, lambda) {
  if (x < 0) {
    return(0)
  } else {
    return(1 - exp(-lambda * x))
  }
}

my_quantile_exp <- function(p, lambda) {
  if (p <= 0 || p >= 1) {
    stop("p 必须在 (0, 1) 区间内")
  }
  return(-log(1 - p) / lambda)
}
lambda <- 1/2
